--[ LIST PROCESSING ]--------------------------------------------------------
Returns the first element of a cons cell
             		                            
(defun car: _.first)
             		           
                               
                            
                 ¶               
               
Returns the last element of a cons cell
             		                            
(defun cdr: _.rest)
             		           
                               
                            
                 ¶               
               
Returns the nth element of a sequence
             		                            
(defun nth: (idx, seq, count = 1) ->
    (seq.slice idx, idx + count))
             		           
                               
                            
                 ¶               
               
Returns the length of a sequence
             		                            
(defun len: _.size)
